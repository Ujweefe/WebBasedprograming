HTML5 Kapsamlı Özet

Bu sayfada HTML5'in temel yapı taşlarını ve sık kullanılan elementlerini bulabilirsiniz.

1. HTML5 Temel Yapı

HTML 
<!DOCTYPE html>
<html lang="tr">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <meta name="description" content="Sayfa açıklaması">
    <title>Sayfa Başlığı</title>
    <link rel="stylesheet" href="style.css">
</head>
<body>
    <header>
        <nav>
            <!-- Navigasyon içeriği -->
        </nav>
    </header>

    <main>
        <article>
            <section>
                <!-- Bölüm içeriği -->
            </section>
        </article>
        
        <aside>
            <!-- Yan içerik -->
        </aside>
    </main>

    <footer>
        <!-- Alt bilgi -->
    </footer>
</body>
</html>

Önemli Notlar

  • DOCTYPE tanımı HTML5 için zorunludur
  • lang özelliği SEO için önemlidir
  • meta viewport mobil uyumluluk için gereklidir
  • Semantik elementler kullanılmalıdır

2. Metin Elementleri

HTML 
<!-- Başlıklar -->
<h1>Ana Başlık</h1>
<h2>Alt Başlık</h2>
<h3>Bölüm Başlığı</h3>

<!-- Paragraflar -->
<p>Normal paragraf metni</p>
<p><strong>Kalın metin</strong></p>
<p><em>İtalik metin</em></p>
<p><mark>Vurgulu metin</mark></p>
<p><code>Kod metni</code></p>

<!-- Alıntılar -->
<blockquote>
    Uzun alıntı metni buraya gelir.
    <cite>- Yazar Adı</cite>
</blockquote>

<!-- Kısa Alıntı -->
<p><q>Kısa alıntı</q></p>

<!-- Alt ve Üst Simge -->
<p>H<sub>2</sub>O ve x<sup>2</sup></p>

4. Linkler ve URL Kullanımı

HTML 
<!-- Temel Link -->
<a href="sayfa.html">Sayfaya Git</a>

<!-- Harici Link -->
<a href="https://www.example.com" target="_blank">Yeni Sekmede Aç</a>

<!-- E-posta Linki -->
<a href="mailto:ornek@mail.com">E-posta Gönder</a>

<!-- Telefon Linki -->
<a href="tel:+901234567890">Ara</a>

<!-- Dosya İndirme Linki -->
<a href="dosya.pdf" download>PDF İndir</a>

<!-- Sayfa İçi Link -->
<a href="#bolum1">Bölüme Git</a>

<!-- Görsel Link -->
<a href="sayfa.html">
    <img src="resim.jpg" alt="Tıklanabilir Resim">
</a>

5. Resimler ve Medya

HTML 
<!-- Temel Resim -->
<img src="resim.jpg" alt="Resim açıklaması">

<!-- Boyutlu Resim -->
<img src="resim.jpg" alt="Resim" width="300" height="200">

<!-- Resim ve Açıklama -->
<figure>
    <img src="resim.jpg" alt="Resim">
    <figcaption>Resim açıklaması</figcaption>
</figure>

<!-- Responsive Resim -->
<picture>
    <source media="(min-width: 800px)" srcset="buyuk.jpg">
    <source media="(min-width: 400px)" srcset="orta.jpg">
    <img src="kucuk.jpg" alt="Responsive resim">
</picture>

<!-- Video -->
<video controls width="500">
    <source src="video.mp4" type="video/mp4">
    Tarayıcınız video elementini desteklemiyor.
</video>

<!-- Ses -->
<audio controls>
    <source src="ses.mp3" type="audio/mpeg">
    Tarayıcınız audio elementini desteklemiyor.
</audio>

7. Formlar

HTML 
<form action="/gonder" method="post">
    <!-- Metin Girişi -->
    <label for="isim">İsim:</label>
    <input type="text" id="isim" name="isim" required>

    <!-- E-posta -->
    <label for="email">E-posta:</label>
    <input type="email" id="email" name="email">

    <!-- Şifre -->
    <label for="sifre">Şifre:</label>
    <input type="password" id="sifre" name="sifre">

    <!-- Sayı -->
    <label for="yas">Yaş:</label>
    <input type="number" id="yas" name="yas" min="0" max="120">

    <!-- Tarih -->
    <label for="tarih">Tarih:</label>
    <input type="date" id="tarih" name="tarih">

    <!-- Onay Kutusu -->
    <label>
        <input type="checkbox" name="onay">
        Kullanım şartlarını kabul ediyorum
    </label>

    <!-- Radio Butonlar -->
    <fieldset>
        <legend>Cinsiyet:</legend>
        <label>
            <input type="radio" name="cinsiyet" value="erkek">
            Erkek
        </label>
        <label>
            <input type="radio" name="cinsiyet" value="kadin">
            Kadın
        </label>
    </fieldset>

    <!-- Açılır Liste -->
    <label for="sehir">Şehir:</label>
    <select id="sehir" name="sehir">
        <option value="">Seçiniz...</option>
        <option value="istanbul">İstanbul</option>
        <option value="ankara">Ankara</option>
        <option value="izmir">İzmir</option>
    </select>

    <!-- Çoklu Seçim -->
    <label for="hobiler">Hobiler:</label>
    <select id="hobiler" name="hobiler[]" multiple>
        <option value="spor">Spor</option>
        <option value="muzik">Müzik</option>
        <option value="kitap">Kitap</option>
    </select>

    <!-- Metin Alanı -->
    <label for="mesaj">Mesaj:</label>
    <textarea id="mesaj" name="mesaj" rows="4"></textarea>

    <!-- Dosya Yükleme -->
    <label for="dosya">Dosya:</label>
    <input type="file" id="dosya" name="dosya">

    <!-- Gönder Butonu -->
    <button type="submit">Gönder</button>
    
    <!-- Sıfırla Butonu -->
    <button type="reset">Sıfırla</button>
</form>
